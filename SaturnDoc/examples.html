Squares


function Square(n)
{
    for (let i = 1; i <= n; i += 1)
    {
        //Get the square of the current number
        Println(i ^ 2);
    }
}

//Ask the user how many squares we want to calculate
Square(StrToNum(Input("Calculate how many squares?")));

FizzBuzz


let fizz = "Fizz";  //What to print for fizz
let buzz = "Buzz";  //What to print for buzz
let targ = 100;     //What index to stop

let ft = 3;         //When to trigger fizz
let bt = 5;         //When to trigger buzz

for(let i = 1; i <= targ; i += 1)
{
    let output = "";

     if (i % ft == 0)
     {
        output = fizz;
     }
     if (i % bt == 0)
     {
        output += buzz;
     }

     //Print output
     if (output == "")
     {
        output = i;
     }

    Println(output);
}

Strings.slf


function ReverseString(s)
{
    let reverse = "";

    let j = 0;
    for (let i = lengthof(s) - 1; i >= 0; i -= 1)
    {
        //This sti command may be confusing, but what it does is,
        //sti will take a string, index and a char. So this call
        //Will get a char from s at index i, and save it to reverse
        //at index j
        //Since variables are passed by value, we then return it to to reverse
        reverse = sti(reverse, j, ldi(s, i));
        j += 1;
    }

    return reverse;
}

function GetChar(string, index) => return ldi(string, index);
