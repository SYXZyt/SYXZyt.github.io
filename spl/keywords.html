let Creates a variable in memory
let x "SPL"
println x
const Creates a non-mutable value in memory
const PI 3.1415
let PI 4 ;This throws an error
print Print text to the screen
println Print tex to the screen, followed by "\r\n"
const NAME "SPL"
print "My name is "
println NAME
goto Jump code execution to a specific point. Note that you can jump to labels or a specific line number
:loop
	goto loop
call Works like goto, but it adds to the callstack, meaning you can return like a function. It also only works on labels, and does not work on line numbers
:loop
	call loop ;Given enough time, this causes a stack overflow
exit Exit the program with a specific error code
exit 100 ;Exit code is 100
push Push a value onto the stack
push x			;Variables can be copied
push "string"	;Strings can be pushed
push 3.1415		;Floats can be pushed
push 0			;Ints can be pushed
pop Remove a value off of the stack
push 0
pop ;Stack size is now 0, as the value added, was removed
setpop Take a value off of the stack, and store it in a variable
push 3.1415
setpop PI ;PI now contains 3.1415
add Add two values together
push 5
push 45
add ;Pushes 50 to the stack
